import random

chars = ['a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j',
         'k', 'l', 'm', 'n', 'o', 'p', 'q', 'r', 's', 't',
         'u', 'v', 'w', 'x', 'y', 'z', 'A' … 'Z',
         '0' … '9', '!', '"', '#', '$', '%', '&', "'",
         '(', ')', '*', '+', ',', '-', '.', '/', ':',
         ';', '<', '=', '>', '?', '@', '[', ']', '^',
         '_', '`', '{', '|', '}', '~']

random.shuffle(chars)
passwd = []
n = 16  # longitud introducida por el usuario

for i in range(n):
    char = random.randint(0, len(chars) - 1)
    passwd.append(chars[char])

passwd_fn = "".join(passwd)
print(passwd_fn)
